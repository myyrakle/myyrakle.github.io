л№ лҘё к°ңл°ң нҷҳкІҪ нқҗлҰ„мқ„ мҷ„м„ұн•ҳкё° мң„н•ҙм„ңлҠ”, л№ лҘҙкІҢ л°°нҸ¬н•ҳкі , л°°нҸ¬к°Җ мҷ„лЈҢлҗҳм—Ҳмқ„л•Ң к·ё мӢңм җмқ„ мһҳ мһЎлҠ” кІғмқҙ мӨ‘мҡ”н•ҳлӢӨ. н•ҳм§Җл§Ң кё°ліё k8s кө¬м„ұм—җм„ңлҠ” мқҙкұё м ңл•Ң мһЎкё°к°Җ мүҪм§ҖлҠ” м•ҠлӢӨ. кі„мҶҚ kubectlлЎң лӘЁлӢҲн„°л§Ғмқ„ н•  мҲҳлҸ„ м—ҶлҠ” л…ёлҰҮмқҙлӢҲ л§җмқҙлӢӨ.

л°°нҸ¬к°Җ мҷ„лЈҢлҗң мӢңм җм—җ Slack л“ұмқ„ нҶөн•ҙм„ң мҷ„лЈҢлҘј нҷ•мӢӨн•ҳкІҢ м•Ң мҲҳ мһҲлӢӨл©ҙ м°ё нҺён•  кІғмқҙлӢӨ.

лӢӨн–үнһҲлҸ„ ArgoCDлҘј м“ё кІҪмҡ°м—җлҠ” мқҙлҹ° л¶Җл¶„мқҙ мўҖ к°„нҺён•ҙм§„лӢӨ. мһҗмІҙм ҒмңјлЎң м•ҢлҰјмҡ© лҸ„кө¬лҘј м§Җмӣҗн•ҳкё° л•Ңл¬ёмқҙлӢӨ.

лЁјм Җ nofiticaion лҰ¬мҶҢмҠӨк°Җ мӢӨн–үлҗҳкі  мһҲлҠ”м§Җ нҷ•мқён•ңлӢӨ.

kubectl get deployment -n argocd 2>/dev/null | grep notification || echo "ArgoCD namespace not found. Checking other namespaces..." && kubectl get deployment -A | grep argocd

ConfigMapмқ„ мһ‘м„ұн•ҳкі 

apiVersion: v1
kind: ConfigMap
metadata:
  name: argocd-notifications-cm
  namespace: argocd  # ArgoCD namespaceлЎң ліҖкІҪ
data:
  # Slack Bot Token м„Өм •
  service.slack: |
    token: Slack нҶ нҒ°

  # Deployment мҷ„лЈҢ н…ңн”ҢлҰҝ
  template.deployment-success: |
    message: |
      вң… *Deployment Completed*
      *Application:* {{.app.metadata.name}}
      {{- if .app.spec.destination.namespace}}
      *Namespace:* {{.app.spec.destination.namespace}}
      {{- end}}
      {{- if .app.spec.destination.server}}
      *Cluster:* {{.app.spec.destination.server}}
      {{- end}}
      *Revision:* {{substr 0 7 .app.status.sync.revision}}
      {{- if .app.status.operationState.operation.initiatedBy.username}}
      *Synced by:* {{.app.status.operationState.operation.initiatedBy.username}}
      {{- end}}
      {{- if .app.status.operationState.syncResult}}
      *Deployed Resources:*
      {{- range .app.status.operationState.syncResult.resources}}
      {{- if or (eq .kind "Deployment") (eq .kind "StatefulSet") (eq .kind "Application")}}
      вҖў {{.kind}}/{{.name}}{{if .namespace}} ({{.namespace}}){{end}}
      {{- end}}
      {{- end}}
      {{- end}}

  # Deployment мӢӨнҢЁ н…ңн”ҢлҰҝ
  template.deployment-failed: |
    message: |
      вқҢ *Deployment Failed*
      *Application:* {{.app.metadata.name}}
      {{- if .app.spec.destination.namespace}}
      *Namespace:* {{.app.spec.destination.namespace}}
      {{- end}}
      {{- if .app.status.operationState.message}}
      *Error:* {{.app.status.operationState.message}}
      {{- end}}
      {{- if .app.status.operationState.syncResult}}
      *Failed Resources:*
      {{- range .app.status.operationState.syncResult.resources}}
      {{- if ne .status "Synced"}}
      вҖў {{.kind}}/{{.name}}{{if .namespace}} ({{.namespace}}){{end}} - {{.status}}
      {{- end}}
      {{- end}}
      {{- end}}

  # Trigger м„Өм •
  trigger.on-deployed: |
    - description: Application is synced and healthy
      send:
      - deployment-success
      when: app.status.operationState.phase in ['Succeeded'] and app.status.health.status == 'Healthy'
    oncePer: app.status.operationState.finishedAt

  trigger.on-deployment-failed: |
    - description: Application sync failed
      send:
      - deployment-failed
      when: app.status.operationState.phase in ['Error', 'Failed']
    oncePer: app.status.operationState.finishedAt

  # кё°ліё кө¬лҸ… м„Өм •
  subscriptions: |
    - recipients:
      - slack:мҠ¬лһҷ мұ„л„җ ID
      triggers:
      - on-deployed
      - on-deployment-failed

к·ёкұё м Ғмҡ©н•ңлӢӨ.

kubectl apply -f argocd-notifications-config.yaml

к·ёлҰ¬кі  кё°мЎҙ Applicationм—җ annotationмқ„ м¶”к°Җн•ҙмӨҖлӢӨ.
м—¬кё°м—җ мҠ¬лһҷ мұ„л„җ IDлҘј л„Јмңјл©ҙ лҗңлӢӨ.

kubectl patch app serverroom-apps -n argocd --type merge -p '{
  "metadata": {
    "annotations": {
      "notifications.argoproj.io/subscribe.on-deployed.slack": "мұ„л„җID",
      "notifications.argoproj.io/subscribe.on-deployment-failed.slack": "мұ„л„җID"
    }
  }
}'

к·ёлҹј мқҙлҹ° мӢқмңјлЎң Deploymentмқҳ л°°нҸ¬к°Җ мҷ„лЈҢлҗ л•Ңл§ҲлӢӨ - м •нҷ•нһҲлҠ” ArgoCD Syncк°Җ лҗ л•Ңл§ҲлӢӨ нҠёлҰ¬кұ°лҗҳм–ҙм„ң м•ҢлҰјмқҙ м „мҶЎлҗ  кІғмқҙлӢӨ.

к·јлҚ° мқҙ л°©мӢқм—җлҠ” мўҖ лӢЁм җмқҙ мһҲлӢӨ. argocdк°Җ Syncм—җ лҢҖн•ң м •ліҙлҘј мғҒм„ён•ҳкІҢ мЈјм§ҖлҠ” м•ҠлҠ”лӢӨлҠ” кІғмқҙлӢӨ. м–ҙл–Ө Deploymentк°Җ ліҖкІҪлҗң кІғмқём§Җ м•Ң мҲҳк°Җ м—ҶлӢӨ.
к·ёлһҳм„ң мқҙкұё мһҳ нҷңмҡ©н•ҳл Өл©ҙ, argocd Applicationмқ„ мӢңмҠӨн…ңлі„лЎң мӘјк°ңм„ң Apps of App нҢЁн„ҙмқ„ мһҳ кө¬нҳ„н•ҙм•ј н•ҳкі  к°ңлі„ App лӢЁмң„м—җ лҢҖн•ҙм„ң NotiлҘј л°ӣлҸ„лЎқ кө¬м„ұн•ҙм•ј н•ңлӢӨ.

к·ёлһҳм„ң кҙҖлҰ¬нҸ¬мқёнҠёк°Җ мўҖ ліөмһЎн•ҙм§Ҳ мҲҳ мһҲлӢӨлҠ” кІғмқҙ лӢЁм җмқҙлӢӨ.
